\section{Software Rangefinder}
	\subsection{Allgemein}
		\begin{itemize}
			\item Software kann auf der Hardware\footnote{\hardwareinfo} debuged werden\dotfill \leer
		\end{itemize}
	\subsection{Infrarot Interrupt}
		\begin{itemize}
			\item Sensoren lösen bei Hindernis Alarm aus:
			\begin{itemize}
				\item Flag für vorne links\dotfill \leer
				\item Flag für vorne rechts\dotfill \leer
				\item Flag für hinten links\dotfill \leer
				\item Flag für hinten rechts\dotfill \leer
			\end{itemize}
			\item Sensoren setzen ohne Hindernis Alarm zurück:
			\begin{itemize}
				\item Flag für vorne links\dotfill \leer
				\item Flag für vorne rechts\dotfill \leer
				\item Flag für hinten links\dotfill \leer
				\item Flag für hinten rechts\dotfill \leer
			\end{itemize}
			\item Software funktioniert auch falls nicht alle Sensoren angeschlossen\dotfill \leer
		\end{itemize}
		%
	\subsection{Ultraschall Task}
		\begin{itemize}
			\item System/FreeRTOS:
			\begin{itemize}
				\item Task wird zu Beginn \textit{suspended}\dotfill \leer
			\end{itemize}
			\item Sensoren lösen bei Hindernis (näher als in Header definiert) Alarm aus:
			\begin{itemize}
				\item Flag für vorne\dotfill \leer
				\item Flag für hinten\dotfill \leer
				\item Flag für Separation\dotfill \leer
			\end{itemize}
			\item Sensoren setzen ohne Hindernis Alarm zurück:
			\begin{itemize}
				\item Sensor vorne\dotfill \leer
				\item Sensor hinten\dotfill \leer
				\item Flag für Separation\dotfill \leer
			\end{itemize}
			\item Verwendung von nur einem Sensor:
			\begin{itemize}
				\item Task funktioniert auch falls nicht beide Sensoren angeschlossen\dotfill \leer
				\item Mit Definition \texttt{RANGEFINDER\_ONLY\_FW} funktioniert nur der vordere Sensor\dotfill \leer
			\end{itemize}
		\end{itemize}
